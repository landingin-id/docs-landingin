import { Step, Steps } from 'fumadocs-ui/components/steps';

<Steps>
  <Step>
  #### Open Produk Link
  Dari halaman produk klik button "Pesan Sekarang" 
  <img src="/uploads/tahapan-order-1.png" alt="Alt text" width="100%" />
  </Step>
  <Step>
  #### Isi Data diri
  Masukan nama, email dan nomor telp lalu klik "Selesaikan Pembayaran"
  <img src="/uploads/tahapan-order-2.png" alt="Alt text" width="100%" />
  </Step>
  <Step>
  #### Pilih Metode Pembayaran
  Masukan nama, email dan nomor telp lalu klik "Selesaikan Pembayaran"
  <img src="/uploads/tahapan-order-3.png" alt="Alt text" width="100%" />
  </Step>
  <Step>
  #### Lakukan Pembayaran
  Lakukan pembayaran sejumlah nominal tertera.
  <img src="/uploads/tahapan-order-4.png" alt="Alt text" width="100%" />
  </Step>
  <Step>
  #### Pembayaran Selesai
  Pembayaran telah selesai, Link download produk otomatis terkirim ke alamat email anda.
  <img src="/uploads/tahapan-order-5.png" alt="Alt text" width="100%" />
  </Step>
</Steps>

<Cards>
    <Card href="#" title="Catatan">
      Untuk opsi metode pembayaran bank transfer, setelah pembeli melakukan konfirmasi pembayaran link download produk
    tidak otomatis terkirim, perlu ada konfirmasi pembayaran terlebih dahulu dari pihak penjual.
    </Card>
</Cards>
